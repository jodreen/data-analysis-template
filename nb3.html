##creating a list of files in the folder
list_of_files <- list.files("~/Desktop/googletrends-stocks/notebooks/data/cleaned/DowJones_Modified")
normal_cor_list <- c()
percent_cor_list <- c()
sell_cor_list <- c()
ticker_cor_list <- c()
buy_cor_list <- c()
my_str = "data/cleaned/DowJones_Modified/"

##creating a function to gather all the data

get_all_data = function(x) {
    if (x == 'SP500') {
        my_str <<- 'data/cleaned/SP500/'
        list_of_files <<- list.files("~/Desktop/googletrends-stocks/notebooks/data/cleaned/SP500")
    }
    for (s in 1:length(list_of_files)){
        ##opening each file in the folder
        myStock = read.csv(paste(my_str,list_of_files[s],sep = ""))
        myStock = na.omit(myStock)
        growth2 = c(0)
        growth = 0
        ##adding a column with relative stock prices
        for (i in 1:length(myStock$Price)){
            growth = (myStock$Price[i]-myStock$Price[i-1])/myStock$Price[i-1]*100
            growth2= c(growth2,growth)
        }
        if (nrow(myStock) == 0){
            next
        }
        if (ncol(myStock) < 3){
            next
        }
        ##getting rid of key term files, so only normal files remain
        if (substr(list_of_files[s],1,4) == "sell"){
            sell_cor_list <<- c(cor(myStock[,2], myStock[,3]),sell_cor_list)
            next
        }
        if (substr(list_of_files[s],1,3) == "buy"){
            buy_cor_list <<- c(cor(myStock[,2], myStock[,3]),buy_cor_list)
            next
        }
        if (toupper(substr(list_of_files[s],2,2)) == (substr(list_of_files[s],2,2))) {
            ticker_cor_list <<- c(cor(myStock[,2], myStock[,3]),ticker_cor_list)
            next
        }
        myStockNew = data.frame(myStock, "PriceGrowth"= growth2)
        ##lists of correlatons for absolute and relative stock price data
        normal_cor_list <<-c(cor(myStockNew[,2], myStockNew[,3]),normal_cor_list)
        percent_cor_list <<-c(cor(myStockNew[,2], myStockNew[,4]),percent_cor_list)
        normal_cor_list <<- na.omit(normal_cor_list)
        percent_cor_list <<- na.omit(percent_cor_list)
        ticker_cor_list <<- na.omit(ticker_cor_list)
        sell_cor_list <<- na.omit(sell_cor_list)
        buy_cor_list <<- na.omit(buy_cor_list)
        }
    return()
}
get_all_data('DowJones_Modified')

get_all_data = function(x) {
    if (x == 'SP500') {
        my_str <<- 'data/cleaned/SP500/'
        list_of_files <<- list.files("~/Desktop/googletrends-stocks/notebooks/data/cleaned/SP500")
    }
    for (s in 1:length(list_of_files)){
        ##opening each file in the folder
        myStock = read.csv(paste(my_str,list_of_files[s],sep = ""))
        myStock = na.omit(myStock)
        growth2 = c(0)
        growth = 0
        ##adding a column with relative stock prices
        for (i in 1:length(myStock$Price)){
            growth = (myStock$Price[i]-myStock$Price[i-1])/myStock$Price[i-1]*100
            growth2= c(growth2,growth)
        }
        ##intermediate data cleaning step which should be unnecessary after cleaning is linked
        if (nrow(myStock) == 0){
            next
        }
        if (ncol(myStock) < 3){
            next
        }
        ##getting rid of key term files, so only normal files remain
        if (substr(list_of_files[s],1,4) == "sell"){
            sell_cor_list <<- c(cor(myStock[,2], myStock[,3]),sell_cor_list)
            next

        if (substr(list_of_files[s],1,3) == "buy"){
            buy_cor_list <<- c(cor(myStock[,2], myStock[,3]),buy_cor_list)
            next

        if (toupper(substr(list_of_files[s],2,2)) == (substr(list_of_files[s],2,2))) {
            ticker_cor_list <<- c(cor(myStock[,2], myStock[,3]),ticker_cor_list)
            next
        }
        myStockNew = data.frame(myStock, "PriceGrowth"= growth2)
